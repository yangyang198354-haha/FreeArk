### 5.1 RagVectorCache（进程内向量缓存）

```python
import threading
import numpy as np
from typing import List, Dict, Optional

class RagVectorCache:
    """进程内向量缓存单例。threading.Lock 保护并发写。"""
    _instance: Optional['RagVectorCache'] = None
    _lock = threading.Lock()

    def __init__(self):
        self._vectors: Optional[np.ndarray] = None  # shape=(N,1024), float32
        self._chunk_ids: List[int] = []
        self._meta: List[Dict] = []             # [{doc_name, source, is_image_ocr}]
        self._rw_lock = threading.Lock()

    @classmethod
    def get_instance(cls) -> 'RagVectorCache':
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = cls()
        return cls._instance

    def load(self) -> None:
        """从 DB 全量加载所有 indexed 文档的 chunks。启动时或刷新时调用。"""
        from .models_rag import RagChunk, RagDocument
        chunks = (RagChunk.objects
                  .filter(document__status='indexed')
                  .select_related('document')
                  .order_by('id'))
        vectors, ids, meta = [], [], []
        for c in chunks:
            vec = np.frombuffer(c.embedding, dtype=np.float32)
            vectors.append(vec)
            ids.append(c.id)
            meta.append({
                'doc_name': c.document.file_name,
                'source': c.page_or_section,
                'is_image_ocr': c.is_image_ocr,
                'content': c.content,
            })
        with self._rw_lock:
            self._vectors = np.array(vectors, dtype=np.float32) if vectors else None
            self._chunk_ids = ids
            self._meta = meta

    def refresh(self) -> None:
        """异步触发重载（在线程中调用 load()）。"""
        t = threading.Thread(target=self.load, daemon=True)
        t.start()

    def search(self, query_vec: np.ndarray, k: int = 5,
               threshold: float = 0.3) -> List[Dict]:
        """余弦相似度 top-k 检索。线程安全只读。"""
        with self._rw_lock:
            if self._vectors is None or len(self._vectors) == 0:
                return []
            vectors = self._vectors
            meta = list(self._meta)
        # 余弦相似度：点积 / (范数乘积)
        query_norm = query_vec / (np.linalg.norm(query_vec) + 1e-9)
        norms = np.linalg.norm(vectors, axis=1, keepdims=True) + 1e-9
        normed = vectors / norms
        scores = normed @ query_norm
        top_idx = np.argsort(scores)[::-1][:k]
        results = []
        for i in top_idx:
            if scores[i] < threshold:
                break
            m = meta[i]
            results.append({
                'content': m['content'],
                'source': f"{m['doc_name']} · {m['source']}",
                'is_image_ocr': m['is_image_ocr'],
                'score': float(scores[i]),
            })
        return results

# 全局单例
rag_vector_cache = RagVectorCache.get_instance()
```

### 5.2 RagEmbedder（向量化）

```python
class RagEmbedder:
    """封装 langchain-openai OpenAIEmbeddings，带 fail-open。"""

    def __init__(self):
        from django.conf import settings
        self._base_url = getattr(settings, 'RAG_EMBEDDING_BASE_URL', '')
        self._model = getattr(settings, 'RAG_EMBEDDING_MODEL', 'BAAI/bge-m3')
        self._api_key = getattr(settings, 'RAG_EMBEDDING_API_KEY', '')

    def _get_embeddings(self):
        from langchain_openai import OpenAIEmbeddings
        return OpenAIEmbeddings(
            base_url=self._base_url,
            model=self._model,
            api_key=self._api_key,
            timeout=5.0,
        )

    def embed_texts(self, texts: List[str]) -> List[np.ndarray]:
        """批量 embed。返回 list[np.ndarray(float32, shape=(1024,))]。"""
        emb = self._get_embeddings()
        vecs = emb.embed_documents(texts)
        return [np.array(v, dtype=np.float32) for v in vecs]

    def embed_query(self, text: str) -> np.ndarray:
        """单次 embed（用于检索 query）。"""
        emb = self._get_embeddings()
        v = emb.embed_query(text)
        return np.array(v, dtype=np.float32)
```

### 5.3 RagParser（文档解析）

```python
from dataclasses import dataclass

@dataclass
class ParsedChunk:
    content: str
    page_or_section: str
    is_image_ocr: bool = False

class RagParser:
    def __init__(self, chunk_size: int = 500, chunk_overlap: int = 50):
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap

    def _split_text(self, text: str, source: str) -> List[ParsedChunk]:
        """按 chunk_size 滑动窗口分块，不在段落中间强切。"""
        chunks = []
        start = 0
        while start < len(text):
            end = start + self.chunk_size
            chunks.append(ParsedChunk(
                content=text[start:end],
                page_or_section=source,
                is_image_ocr=False,
            ))
            start += self.chunk_size - self.chunk_overlap
        return chunks

    def parse_docx(self, file_bytes: bytes) -> List[ParsedChunk]:
        """python-docx 解析 .docx，提取段落文字+图片。"""
        import io
        from docx import Document
        doc = Document(io.BytesIO(file_bytes))
        chunks = []
        para_idx = 0
        text_buffer = ""
        for para in doc.paragraphs:
            para_idx += 1
            text = para.text.strip()
            if text:
                text_buffer += text + "\n"
                if len(text_buffer) >= self.chunk_size:
                    chunks.extend(self._split_text(text_buffer, f"段落 {para_idx}"))
                    text_buffer = text_buffer[-self.chunk_overlap:] if self.chunk_overlap else ""
        if text_buffer.strip():
            chunks.append(ParsedChunk(content=text_buffer.strip(),
                                      page_or_section=f"段落 {para_idx}"))
        # 图片 OCR
        for shape_idx, rel in enumerate(doc.part.rels.values()):
            if "image" in rel.target_ref:
                img_bytes = rel.target_part.blob
                ocr_text = self._ocr_image(img_bytes)
                if ocr_text:
                    chunks.append(ParsedChunk(
                        content=ocr_text,
                        page_or_section=f"图片 {shape_idx + 1}",
                        is_image_ocr=True,
                    ))
        return chunks

    def parse_pdf(self, file_bytes: bytes) -> List[ParsedChunk]:
        """PyMuPDF 解析 .pdf，提取页面文字+图片。"""
        import io
        import fitz  # PyMuPDF
        doc = fitz.open(stream=io.BytesIO(file_bytes), filetype="pdf")
        chunks = []
        for page_num, page in enumerate(doc, start=1):
            text = page.get_text().strip()
            if text:
                chunks.extend(self._split_text(text, f"第 {page_num} 页"))
            # 图片 OCR
            for img_idx, img_info in enumerate(page.get_images(full=True)):
                xref = img_info[0]
                base_image = doc.extract_image(xref)
                img_bytes = base_image["image"]
                ocr_text = self._ocr_image(img_bytes)
                if ocr_text:
                    chunks.append(ParsedChunk(
                        content=ocr_text,
                        page_or_section=f"第 {page_num} 页 图片{img_idx + 1}",
                        is_image_ocr=True,
                    ))
        return chunks

    @staticmethod
    def _ocr_image(img_bytes: bytes) -> str:
        """OCR 单张图片，失败时返回空串（跳过，不抛出）。"""
        try:
            from rapidocr_onnxruntime import RapidOCR
            ocr = RapidOCR()
            result, _ = ocr(img_bytes)
            if result:
                return "\n".join([line[1] for line in result if line[1]])
        except Exception as e:
            import logging
            logging.getLogger("api.rag_service").warning(
                "OCR 失败，跳过图片: %s", e)
        return ""
```

### 5.4 RagIngestor（入库调度）

```python
class RagIngestor:
    """后台线程入口：解析→分块→向量化→写DB→刷缓存。"""

    def ingest(self, doc_id: int, file_bytes: bytes, file_ext: str) -> None:
        """在后台线程中运行。捕获所有异常写 failed 状态。"""
        from .models_rag import RagDocument, RagChunk
        from django.conf import settings
        try:
            doc = RagDocument.objects.get(id=doc_id)
        except RagDocument.DoesNotExist:
            return  # 文档已被删除，安全退出

        try:
            # 1. 更新状态为 parsing
            doc.status = 'parsing'
            doc.save(update_fields=['status', 'updated_at'])

            # 2. 解析文档
            chunk_size = getattr(settings, 'RAG_CHUNK_SIZE', 500)
            chunk_overlap = getattr(settings, 'RAG_CHUNK_OVERLAP', 50)
            parser = RagParser(chunk_size=chunk_size, chunk_overlap=chunk_overlap)
            if file_ext == '.docx':
                parsed = parser.parse_docx(file_bytes)
            elif file_ext == '.pdf':
                parsed = parser.parse_pdf(file_bytes)
            else:
                raise ValueError(f"不支持的文件类型: {file_ext}")

            if not parsed:
                raise ValueError("文档解析后无任何内容")

            # 检查文档是否仍存在（可能被并发删除）
            if not RagDocument.objects.filter(id=doc_id).exists():
                return  # 安全退出

            # 3. 向量化（批量，每批 20 个 chunk）
            embedder = RagEmbedder()
            BATCH_SIZE = 20
            all_chunks_data = []
            for i in range(0, len(parsed), BATCH_SIZE):
                batch = parsed[i:i+BATCH_SIZE]
                texts = [c.content for c in batch]
                vecs = embedder.embed_texts(texts)
                for chunk, vec in zip(batch, vecs):
                    all_chunks_data.append((chunk, vec))

            # 4. 写入 DB（再次检查文档是否存在）
            if not RagDocument.objects.filter(id=doc_id).exists():
                return
            chunk_objs = []
            for idx, (chunk, vec) in enumerate(all_chunks_data):
                chunk_objs.append(RagChunk(
                    document_id=doc_id,
                    chunk_index=idx,
                    content=chunk.content,
                    embedding=vec.tobytes(),
                    page_or_section=chunk.page_or_section,
                    is_image_ocr=chunk.is_image_ocr,
                ))
            RagChunk.objects.bulk_create(chunk_objs)

            # 5. 更新状态为 indexed
            doc.status = 'indexed'
            doc.chunk_count = len(chunk_objs)
            doc.error_message = ''
            doc.save(update_fields=['status', 'chunk_count', 'error_message', 'updated_at'])

            # 6. 刷新内存缓存
            rag_vector_cache.refresh()

        except Exception as e:
            import traceback
            error_text = f"{type(e).__name__}: {e}"
            try:
                RagDocument.objects.filter(id=doc_id).update(
                    status='failed',
                    error_message=error_text[:1000],
                )
            except Exception:
                pass  # DB 也挂了，无能为力
            import logging
            logging.getLogger("api.rag_service").error(
                "文档 %s 入库失败: %s\n%s", doc_id, error_text,
                traceback.format_exc())
```

### 5.5 search_sanheng_knowledge 实现（在 fa_tools.py 中调用）

```python
# rag_service.py 提供的检索入口
def search_rag(query: str, k: int = 5, threshold: float = 0.3) -> dict:
    """
    对外检索接口。fail-open：任何异常返回 {"chunks":[],"degraded":True}。
    空库返回 {"chunks":[],"degraded":False}。
    """
    try:
        embedder = RagEmbedder()
        query_vec = embedder.embed_query(query)
        results = rag_vector_cache.search(query_vec, k=k, threshold=threshold)
        return {"chunks": results, "degraded": False}
    except Exception as e:
        import logging
        logging.getLogger("api.rag_service").warning(
            "RAG 检索失败（降级）: %s", e)
        return {"chunks": [], "degraded": True}
```

@tool
def search_sanheng_knowledge(query: str) -> str:
    """在三恒知识库中检索与 query 相关的文档片段，用于辅助原理/参数/故障码解答。
    返回最相关的 chunk 文本列表及来源；库为空或不可达时返回空列表提示（不报错）。"""

    from api.rag_service import search_rag

    threshold = getattr(settings, 'RAG_SCORE_THRESHOLD', 0.3)
    result = search_rag(query, k=k, threshold=threshold)
    if result.get('degraded'):
        return "[知识库暂时不可达，以下为通用知识参考。degraded=true]"
    chunks = result.get('chunks', [])
    if not chunks:
        return "[知识库中未找到与该问题相关的内容]"
    lines = [f"[检索到 {len(chunks)} 条相关内容]"]
    for i, c in enumerate(chunks, 1):
        src_note = "（图片OCR）" if c['is_image_ocr'] else ""
        lines.append(
            f"\n[{i}] 来源: {c['source']}{src_note}\n    {c['content'][:400]}"
        )
    return "\n".join(lines)

router.beforeEach((to, from, next) => {
  const requiresAuth = to.matched.some(record => record.meta.requiresAuth)
  const requiresAdmin = to.matched.some(record => record.meta.requiresAdmin)
  const isLoggedIn = localStorage.getItem('userToken') !== null

  if (requiresAuth && !isLoggedIn) {
    next({ name: 'Login' })
    return
  }
  if (requiresAdmin) {
    const userInfo = JSON.parse(localStorage.getItem('userInfo') || '{}')
    if (!userInfo.is_staff) {
      next({ path: '/home' })
      return
    }
  }
  next()
})
